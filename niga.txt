#include <iostream>
using namespace std;

int main() {
    int n;
    do {
        cout << "Podaj ilosc elementow tablicy a: " << endl << "n = ";
        cin >> n;
        if (n > 10) {
            cout << "Pamietaj ze ilosc podanych elementow tablicy a[] nie przekroczyla 10!" << endl;
        }
    } while (n > 10);

    int a[10];
    for (int i = 0; i < n; i++) {
        cout << "Podaj a[" << i << "] = ";
        cin >> a[i];
    }

    int zmiany = 0;
    for (int i = 0; i < n - 1; i++) {
        if ((a[i] >= 0 && a[i + 1] < 0) || (a[i] < 0 && a[i + 1] >= 0)) {
            zmiany++;
        }
    }

    cout << "Ilosc zmian znakow: " << zmiany << endl;
    return 0;
}
